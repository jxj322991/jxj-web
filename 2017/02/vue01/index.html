动态数据绑定:如何监听其属性的读取与变化？
也就是说，如何知道程序访问了对象的哪个属性，又改变了哪个属性？
知识点 : getter/setter/defineProperty

问题

下面的例子如何实现?
let app1 = new Vue({
  name: 'youngwind',
  age: 25
});

let app2 = new Vue({
  university: 'bupt',
  major: 'computer'
});
要实现的结果如下：
app1.data.name 
// alert你访问了 name
app1.data.age = 100;  
// alert你设置了 age，新的值为100
app2.data.university 
// alert你访问了 university
app2.data.major = 'science'  
// alert你设置了 major，新的值为 science

实验一下

原理

源码

function Vue(data) {
    this.data = data;
    this.walk(data)
}

let p = Vue.prototype;

p.walk = function(obj) {
    let val;
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            val = obj[key];
            if (typeof val === 'object') {
                new Vue(val);
            }
            this.convert(key, val);
        }
    }
};

p.convert = function(key, val) {
    Object.defineProperty(this.data, key, {
        enumerable: true,
        configurable: true,
        get: function() {
            alert('你访问了' + key);
            return val
        },
        set: function(newVal) {
            alert('你设置了' + key + ',新的值为' + newVal);
            if (newVal === val) return;
            val = newVal
        }
    })
};

let app1 = new Vue({
    name: 'youngwind',
    age: 25
});

let app2 = new Vue({
    university: 'bupt',
    major: 'computer'
});
